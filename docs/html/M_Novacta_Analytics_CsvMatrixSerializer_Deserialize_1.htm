Remarks

This method reads CSV files created by method Serialize(String, DoubleMatrix) or one of its overloaded versions.

Examples

In the following example, the content of a well formed CSV document is deserialized to create a matrix instance. In its first line, the document contains information about the matrix number of rows and columns, and optionally about the matrix name. The second line contains the optional column names. Additional lines contain information about the matrix rows. For a given row, the individual name is optional and is signaled by a non null string in the first token; subsequent tokens are interpreted as string representations of matrix entries.

Deserializing a CSV document to create a matrix
Copy
using System;
using System.IO;

namespace Novacta.Analytics.CodeExamples
{
    public class CsvSerializeExample0  
    {
        public void Main()
        {
            // Create the CSV representation of a dense matrix.
            string[] data = new string[4] {
                "Dense,2,3,MatrixName",
                ",column0,,column2",
                "row0,1.0,NaN,5.0",
                "row1,2.0,4.0,6.0"};

            // Show the matrix CSV representation.
            Console.WriteLine("CSV representation of the matrix:");
            for (int i = 0; i < data.Length; i++) {
                Console.WriteLine(data[i]);
            }
            Console.WriteLine();

            // Create a stream containing the CSV content.
            MemoryStream stream = new MemoryStream();
            StreamWriter writer = new StreamWriter(stream);
            for (int i = 0; i < data.Length; i++) {
                writer.WriteLine(data[i].ToCharArray());
                writer.Flush();
            }
            stream.Position = 0;

            // Create a reader for the stream.
            StreamReader reader = new StreamReader(stream);

            // Deserialize the CSV document contained in the stream.
            DoubleMatrix matrix = CsvMatrixSerializer.Deserialize(reader);

            // Show the matrix.
            Console.WriteLine("Deserialized matrix:");
            Console.WriteLine(matrix);
        }
    }
}

// Executing method Main() produces the following output:
// 
// CSV representation of the matrix:
// Dense,2,3,MatrixName
// ,column0,,column2
// row0,1.0,NaN,5.0
// row1,2.0,4.0,6.0
// 
// Deserialized matrix:
//                  [column0]                         [column2]        
// [row0]           1                NaN              5                
// [row1]           2                4                6                
//
